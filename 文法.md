```java
CompUnit   -> FuncDef
FuncDef    -> FuncType Var '(' ')' Block
FuncType   -> 'int'
Var      -> 'main'
Block      -> '{' Stmt '}'
Stmt       -> 'return' Exp ';'
Exp        -> AddExp
AddExp     -> MulExp 
              | AddExp ('+' | '−') MulExp
MulExp     -> UnaryExp
              | MulExp ('*' | '/' | '%') UnaryExp
UnaryExp   -> PrimaryExp | ('+' | '-') UnaryExp
PrimaryExp -> '(' Exp ')' | Num|LVal
// UnaryOp    -> '+' | '-'
```
